5 -> 0b101 = 1 * 2^0 + 0 * 2^1 + 1 * 2^2 + ...
4 -> 0b100 = 0 * 2^0 + 0 * 2^1 + 1 * 2^2 + ...

-1  -> - 0b1

a \in int8  | --> (a) + (-a) = 0
-a \in int8 |

a = 0b11011
-a = 0 - a = 0b00..0 - 0b1101 = ~a + 1
a + (~a) = 0b1...1
a + (~a) + 1 = 0b1..1 + 1 = 0
~a + 1 = -a

0b0001 + 0b1110 = 0b1111
